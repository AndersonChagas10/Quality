	Id int not null identity primary key,
	AddDate Date not null,
	AlterDate Date,
	ItemMenu_Id int not null FOREIGN KEY REFERENCES ItemMenu(Id),
	Elaborador_Id int not null FOREIGN key REFERENCES UserSGQ(Id),

	ParLevel1_Id int not null FOREIGN key REFERENCES ParLevel1(Id),
	ParCompany_Id int FOREIGN key REFERENCES UserSGQ(Id),

)
